Problem 1 : Article Views I

sol 1:

import pandas as pd

def article_views(views: pd.DataFrame) -> pd.DataFrame:
    df= views[views['author_id']==views['viewer_id']]
    df = df.drop_duplicates(subset=['author_id'], inplace=False)
    df.sort_values(by =['author_id'],inplace=True)
    df = df[['author_id']].rename(columns={'author_id': 'id'})
    return df


sol 2:

import pandas as pd

def article_views(views: pd.DataFrame) -> pd.DataFrame:
    df= views[views['author_id']==views['viewer_id']]
    df = df['author_id'].unique()
    df=pd.DataFrame(df,columns = ['id'])
    return df.sort_values(by =['id'])

import pandas as pd

def invalid_tweets(tweets: pd.DataFrame) -> pd.DataFrame:
    is_valid = tweets['content'].str.len() > 15
    df = tweets[is_valid]
    return df[['tweet_id']]
